import cv2
import numpy as np
import os

input_image_file = 'Screenshot_20251022-225028.png'
output_image_file = 'grayscale_screenshot.png'

# Read the image
img = cv2.imread(input_image_file, cv2.IMREAD_UNCHANGED)

# Convert to Grayscale
if len(img.shape) >= 3:
    if img.shape[2] == 4:
        img_bgr = cv2.cvtColor(img, cv2.COLOR_BGRA2BGR)
        gray_img = cv2.cvtColor(img_bgr, cv2.COLOR_BGR2GRAY)
    else:
        gray_img = cv2.cvtColor(img, cv2.COLOR_BGR2GRAY)
else:
    gray_img = img.copy()

# Save the grayscale image
cv2.imwrite(output_image_file, gray_img)
